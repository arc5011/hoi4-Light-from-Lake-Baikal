			if = {
				limit = {
					check_variable = { 
						var = cpc_kmt_balance
						value = 3
						compare = greater_than_or_equals
					}
				}
				country_event = { id = LFBUG.101 days = 1 }
			}
			if = {
				limit = {
					check_variable = { 
						var = cpc_kmt_balance
						value = -3
						compare = less_than_or_equals
					}
				}
				country_event = { id = LFBUG.201 days = 1 }
			}
			if = {
				limit = {
					AND = {
						check_variable = { 
							var = cpc_kmt_balance
							value = 2
							compare = less_than_or_equals
						}
						check_variable = { 
							var = cpc_kmt_balance
							value = -2
							compare = greater_than_or_equals
						}
					}
				}
				country_event = { id = LFBUG.10 days = 1 }
			}
		}
	}
